Lab: Table Pattern

The Table pattern displays data in rows and columns with clickable column headers for sorting and numbered pagination buttons at the bottom.

Step 1 — HTML with Hardcoded Content

Write the full table page with real hardcoded rows and pagination buttons — no JavaScript. You need a <table> with <thead> header columns and <tbody> data rows, plus a pagination bar below.

Step 3 — Dynamic Rendering with JavaScript

Remove the hardcoded <tr> rows and the hardcoded page buttons. Leave <tbody> and #page-numbers empty, then add a <script> tag before </body>.

Interactive idea: keep sorting and pagination state in variables, then call one render function after every interaction.

3a — State variables

let allPokemon    = [];
let currentPage   = 1;
let itemsPerPage  = 10;
let sortColumn    = 'id';
let sortDirection = 'asc';

3b — Render the table

function renderTable() {
  // 1) Create a sorted copy (do not mutate source array).
  let sorted = [...allPokemon].sort((a, b) => {
    let valA = a[sortColumn];
    let valB = b[sortColumn];
    if (typeof valA === 'string') valA = valA.toLowerCase();
    if (typeof valB === 'string') valB = valB.toLowerCase();
    if (valA < valB) return sortDirection === 'asc' ? -1 : 1;
    if (valA > valB) return sortDirection === 'asc' ?  1 : -1;
    return 0;
  });

  // 2) Slice only rows for the current page.
  const start    = (currentPage - 1) * itemsPerPage;
  const pageData = sorted.slice(start, start + itemsPerPage);

  // 3) Convert page rows to HTML.
  let html = '';
  for (let p of pageData) {
    html += `<tr>
      <td>${p.id}</td>
      <td>${p.name}</td>
      <td>${p.type1}${p.type2 ? ' / ' + p.type2 : ''}</td>
      <td>${p.hp}</td>
      <td>${p.attack}</td>
      <td>${p.defense}</td>
      <td>${p.speed}</td>
    </tr>`;
  }
  document.querySelector('#table-body').innerHTML = html;

  // 4) Refresh helper UI.
  updateSortIndicators();
  updatePagination();
}

3c — Sorting

function sortBy(column) {
  // Toggle direction when clicking same column.
  if (sortColumn === column) {
    sortDirection = sortDirection === 'asc' ? 'desc' : 'asc';
  } else {
    // New column starts in ascending order.
    sortColumn    = column;
    sortDirection = 'asc';
  }
  // Jump back to first page after sorting.
  currentPage = 1;
  renderTable();
}

function updateSortIndicators() {
  // Keep only the active arrow on the active column.
  document.querySelectorAll('thead th').forEach(th => {
    th.classList.remove('sort-asc', 'sort-desc');
    if (th.dataset.col === sortColumn) {
      th.classList.add(sortDirection === 'asc' ? 'sort-asc' : 'sort-desc');
    }
  });
}

Update the <thead> to add data-col attributes and onclick handlers:

<thead>
  <tr>
    <th data-col="id"      onclick="sortBy('id')">ID</th>
    <th data-col="name"    onclick="sortBy('name')">Name</th>
    <th>Type</th>
    <th data-col="hp"      onclick="sortBy('hp')">HP</th>
    <th data-col="attack"  onclick="sortBy('attack')">Attack</th>
    <th data-col="defense" onclick="sortBy('defense')">Defense</th>
    <th data-col="speed"   onclick="sortBy('speed')">Speed</th>
  </tr>
</thead>

3d — Pagination

function updatePagination() {
  const totalPages = Math.ceil(allPokemon.length / itemsPerPage);
  const prevBtn    = document.querySelector('#prev-btn');
  const nextBtn    = document.querySelector('#next-btn');
  prevBtn.disabled = currentPage === 1;
  nextBtn.disabled = currentPage === totalPages;

  let html = ''; // numbered page buttons
  for (let i = 1; i <= totalPages; i++) {
    html += `<button class="page-btn ${i === currentPage ? 'active' : ''}"
                     onclick="goToPage(${i})">${i}</button>`;
  }
  document.querySelector('#page-numbers').innerHTML = html;
}

function goToPage(page) {
  const totalPages = Math.ceil(allPokemon.length / itemsPerPage);
  // Ignore invalid page requests.
  if (page < 1 || page > totalPages) return;
  currentPage = page;
  renderTable();
}

Bind Previous and Next directly in HTML:

<button id="prev-btn" onclick="goToPage(currentPage - 1)">Previous</button>
<button id="next-btn" onclick="goToPage(currentPage + 1)">Next</button>

3e — Load data

async function loadPokemon() {
  // Fetch dataset once, then render progressively.
  for (let i = 1; i <= 50; i++) {
    const res  = await fetch(`https://pokedextr.uwista.dev/pokemon/${i}`);
    const data = await res.json();
    allPokemon.push(data);
    if (i % 10 === 0) renderTable();
  }
  renderTable();
}

loadPokemon();

3f — Interactive flow (what runs on each action)

  • Click column header: sortBy(column) updates sortColumn/sortDirection, resets page to 1, then calls renderTable().
  • Click page button (or HTML-bound Next/Previous): goToPage(page) updates currentPage, then calls renderTable().
  • renderTable() always does the same pipeline: sort array → slice current page → render rows → refresh pagination UI.
The table now sorts on any column header click and paginates through all 50 entries.
