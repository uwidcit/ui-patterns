Lab: Browse Pattern

The Browse pattern lets users page through a large collection and bookmark items they want to keep. The page is split: the main area shows paginated cards, and a sidebar shows the bookmarks list. Bookmarks are stored in a plain JavaScript array — no database or localStorage needed.

Step 1 — HTML with Hardcoded Content

Write the full two-column layout with hardcoded cards, a hardcoded bookmark entry, and pagination buttons — no JavaScript. You need a #content-area for cards, a #bookmarks sidebar, and a .pagination bar outside <main>.

Step 3 — Dynamic Rendering with JavaScript

Remove all hardcoded cards and bookmark items. Leave #content-area and #bookmarks-list empty, clear the hardcoded page buttons from #page-numbers, then add a <script> tag before </body>.

Interactive idea: maintain two arrays (allPokemon and bookmarks). Every user action updates one array, then re-renders the affected UI blocks.

3a — State variables

let allPokemon  = [];
let bookmarks   = [];    // plain array — no localStorage
let currentPage  = 1;
let itemsPerPage = 12;

3b — Render cards

function renderContent() {
  const area  = document.querySelector('#content-area');
  // Calculate current page slice.
  const start = (currentPage - 1) * itemsPerPage;
  const page  = allPokemon.slice(start, start + itemsPerPage);

  // Build card HTML from page data.
  let html = '';
  for (let p of page) {
    // Check if this card is already bookmarked.
    const saved = bookmarks.some(b => b.id === p.id);
    html += `<div class="card">
      <img src="${p.image}" alt="${p.name}">
      <p><strong>${p.name}</strong></p>
      <p>${p.type1}${p.type2 ? ' / ' + p.type2 : ''}</p>
      <button onclick="toggleBookmark(${p.id})">
        ${saved ? '★ Remove' : '☆ Save'}
      </button>
    </div>`;
  }

  area.innerHTML = html;
  // Keep paging UI in sync with current page.
  updatePagination();
}

3c — Bookmarking

function toggleBookmark(id) {
  // Get clicked object and check if saved.
  const pokemon = allPokemon.find(p => p.id === id);
  const index   = bookmarks.findIndex(b => b.id === id);

  if (index === -1) {
    // Not saved yet: add it.
    bookmarks.push(pokemon);
  } else {
    // Already saved: remove it.
    bookmarks.splice(index, 1);
  }

  // Re-render both main list and sidebar.
  renderContent();
  renderBookmarks();
}

function removeBookmark(id) {
  // Remove one item from bookmark array.
  bookmarks = bookmarks.filter(b => b.id !== id);
  renderContent();
  renderBookmarks();
}

function clearBookmarks() {
  // Reset to empty list.
  bookmarks = [];
  renderContent();
  renderBookmarks();
}

function renderBookmarks() {
  const list  = document.querySelector('#bookmarks-list');
  const count = document.querySelector('#bookmark-count');
  // Reflect bookmark count in the header.
  count.textContent = `${bookmarks.length} Bookmarked`;

  if (bookmarks.length === 0) {
    list.innerHTML = '<p>None yet.</p>';
    return;
  }

  let html = ''; // sidebar item markup
  for (let p of bookmarks) {
    html += `<div class="bookmark-item">
      <img src="${p.image}" alt="${p.name}">
      <span style="text-transform:capitalize">${p.name}</span>
      <button onclick="removeBookmark(${p.id})">Remove</button>
    </div>`;
  }
  list.innerHTML = html;
}

3d — Pagination

function updatePagination() {
  // Total pages depends on dataset size.
  const total   = Math.ceil(allPokemon.length / itemsPerPage);
  const prevBtn = document.querySelector('#prev-btn');
  const nextBtn = document.querySelector('#next-btn');

  prevBtn.disabled = currentPage === 1;
  nextBtn.disabled = currentPage === total;

  let html = ''; // numbered page buttons
  for (let i = 1; i <= total; i++) {
    html += `<button class="page-btn ${i === currentPage ? 'active' : ''}"
                     onclick="goToPage(${i})">${i}</button>`;
  }
  document.querySelector('#page-numbers').innerHTML = html;
}

function goToPage(page) {
  const total = Math.ceil(allPokemon.length / itemsPerPage);
  // Ignore invalid page clicks.
  if (page < 1 || page > total) return;
  currentPage = page;
  renderContent();
}

Bind Previous and Next directly in HTML:

<button id="prev-btn" onclick="goToPage(currentPage - 1)">Previous</button>
<button id="next-btn" onclick="goToPage(currentPage + 1)">Next</button>

3e — Load data

async function loadPokemon() {
  // Fetch dataset and render progressively.
  for (let i = 1; i <= 50; i++) {
    const res  = await fetch(`https://pokedextr.uwista.dev/pokemon/${i}`);
    const data = await res.json();
    allPokemon.push(data);
    if (i % 12 === 0 || i === 1) renderContent();
  }
  renderContent();
  renderBookmarks();
}

loadPokemon();

3f — Interactive flow (what runs on each action)

  • Click Save/Remove on a card: toggleBookmark(id) updates bookmarks, then calls renderContent() and renderBookmarks().
  • Click Remove/Clear All in sidebar: update bookmarks, then re-render both content and sidebar.
  • Click page buttons: goToPage(page) updates currentPage, then calls renderContent() (which also refreshes pagination).
Clicking ☆ Save on any card adds it to the bookmarks panel. The ★ Remove button and the sidebar Remove link both update the same bookmarks array — no page reload needed.
